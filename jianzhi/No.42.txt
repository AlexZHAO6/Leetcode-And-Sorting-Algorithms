    public int maxSubArray(int[] nums) {
    	if(nums.length == 0)
    		return 0;
    	if(nums.length == 1)
    		return nums[0];
    	int len = nums.length;
    	int[] dp = new int[len];
    	dp[0] = nums[0];
    	int max = dp[0];
    	for(int i = 1; i < len; i++) {
    		if(dp[i-1] > 0)
    			dp[i] = dp[i-1] + nums[i];
    		else
    			dp[i] = nums[i];
    		
    		max = Math.max(max, dp[i]);
    		
    	}
    	
    	return max;

    } //dpï¼
