    public String longestPalindrome(String s) {
    	int len = s.length();
        if(len <= 1)
            return s;
    	boolean[][] dp = new boolean[len][len];
    	
    	String ans = s.substring(0,1);
    	for(int r = 1; r < len; r++) {
    		for(int l = 0; l < r; l++) {
    			if(s.charAt(r) == s.charAt(l) && (r - l <= 2 || dp[l+1][r-1]))
    				dp[l][r] = true;

    			if(dp[l][r] && r - l + 1 > ans.length()) {
    				ans = s.substring(l, r+1);

    			}
    				
    			
    			
    		}
    	}
    	
    	return ans;
    	       
    } //dp解决
