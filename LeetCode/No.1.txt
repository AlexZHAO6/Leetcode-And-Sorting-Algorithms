    public int[] twoSum(int[] nums, int target) {
    	int n = nums.length;
    	Map<Integer,Integer> map = new HashMap<>();
    	for(int i = 0; i < n; i++)
    		map.put(nums[i], i);
    	
    	for(int i = 0; i < n; i++) {
    		int remain = target - nums[i];
    		
    		if(map.containsKey(remain) && map.get(remain) != i)
    			return new int[] {i,map.get(remain)};
    	}
    	
    	return new int[2];
     
    }
